#--- Auxiliary functions ---
#--- Lead/Lag operator ---
shift <- function(x, n_periods) {
  if (n_periods == 0) return(x);
  
  if (n_periods > 0) {
    out <- c(x[(n_periods + 1):length(x)], rep(NA, n_periods));
    return(out);
  }
  
  if (n_periods < 0) {
    out <- c(rep(NA, -n_periods), x[1:(length(x) + n_periods)]);
    return(out);
  }
}

#--- One-sided Hodrick-Prescott filter ---
hp1 <- function(x, lambda = 1600L) {
  dx.init <- x[2] - x[1];
  
  return (sapply(1:length(x), 
                 function(t) {
                   return (hp2(c(x[1] - 2*dx.init, 
                                 x[1] - dx.init, 
                                 x[1:t]), lambda)[t]);
                   }));
}

#--- Two-sided Hodrick-Prescott filter ---
hp2 <- function(x, lambda = 1600L) {
  m <- length(x);
  E <- diag(nrow = m, ncol = m);
  
  D <- cbind(0L, diag(nrow = m - 2L, ncol = m - 2L)) - 
    cbind(diag(nrow = m - 2L, ncol = m - 2L), 0L);
  D <- cbind(0L, D) - cbind(D, 0L);
  
  return(solve(E + lambda * t(D) %*% D, x));
}

#---GMM estimation---
library(gmm);

# Inflation
g.data <- pc.data[c("inflation")];
g.data <- cbind(g.data, shift(g.data$inflation, 1));

# Economic activity
g.data <- cbind(g.data, pc.data$MC);

# Moment conditions - instruments
g.data <- cbind(g.data, shift(g.data$inflation, -1));
g.data <- cbind(g.data, shift(g.data$inflation, -2));
g.data <- cbind(g.data, shift(g.data$inflation, -3));
g.data <- cbind(g.data, shift(g.data$inflation, -4));

g.data <- cbind(g.data, pc.data$lab.share);
g.data <- cbind(g.data, pc.data$output.gap);

g.data <- cbind(g.data, pc.data$inflation.other);

# Instrument count
instr.count <- ncol(g.data) - 3; # first 3 vars are in the NK PC equation
colnames(g.data) <- c("infl", "f.infl", "activity", 
                      sapply(1:instr.count, function(i) return(paste0("instr.", i))));
g.data <- g.data[complete.cases(g.data),];

g <- function(theta, x) {
  # Errors in the NK PC equation
  err <- x$infl - theta[1] * x$f.infl - theta[2] * x$activity;
  
  # Evaluated moments conditions
  m <- lapply(1:instr.count, 
              function(i) return(err * x[[paste0("instr.", i)]]));
  return(do.call(cbind, m));
}

Jg <- function(theta, x) {
  # Jacobian of the moment conditions
  J <- matrix(c(sapply(1:instr.count, 
                       function(i) return(c(-mean(x$f.infl * x[[paste0("instr.", i)]]), 
                                            -mean(x$activity * x[[paste0("instr.", i)]]))))), 
              nrow = instr.count, ncol = 2, byrow = T);
  return(J);
}

pc.model <- gmm(g = g, x = g.data, 
                t0 = c(beta = 1, kappa = 0), gradv = Jg);
summary(pc.model);
